	constructor (maximo: INTEGER_32)
		local
			arr: ARRAY [INTEGER_32]
			i: INTEGER_32
		do
			create vertices.make_empty

			from
				i := 1
			until
				i > maximo
			loop
				create arr.make_filled (0, 1, maximo)
				matriz.force (arr, i)
				i := i + 1
			end
			numver := 0
		end

		local

	dijktra (s: INTEGER_32; f: INTEGER_32)
		local
			visto: ARRAY [BOOLEAN]
			distancia: ARRAY [INTEGER_32]
			i: INTEGER_32
			v: INTEGER_32
			vis: BOOLEAN
			j: INTEGER_32
			n: INTEGER_32
		do
			create visto.make_filled (False, 1, numver)
			create distancia.make_filled (0, 1, numver)
			from
				i := 1
			until
				i > numver
			loop
				if (not adyacente (s, i)) then
					distancia [i] := -1
				else
					distancia [i] := matriz.item (s).item (i)
				end
				i := i + 1
			end
			distancia [s] := 0
			visto [s] := True
			from
				n := 0
			until
				n = 1
			loop
				v := minimo (visto, distancia)
				visto [v] := True
				from
					j := 1
				until
					j > numver
				loop
					if (distancia [j] > (distancia [v] + matriz.item (v).item (j))) then
						distancia [j] := distancia [v] + matriz.item (v).item (j)
					end
					j := j + 1
				end
				if (todov (visto)) then
					n := 1
				end
			end
			print (">>EL PESO DE LA RUTA MAS CORTA ES: " + distancia [f].out)
			print ("%N>>RECORRIDO:")
		end

			end

		end

					ban := False
				end
				i := i + 1
			end

		end
